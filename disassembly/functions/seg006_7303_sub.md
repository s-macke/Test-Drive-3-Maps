; sub_15153 — load-time polygon decoder.
;
; Expands an on-disk 3D object's 8-byte-per-polygon block into the engine's
; 10-byte-per-polygon runtime format. Called from sub_14FC4 (via sub_12D67
; and other paths) once per object as it becomes visible.
;
; Inputs at entry (informal — most are global state):
;   es:si        = source on-disk polygon stream (8 bytes per record)
;   word_2D8A8   = destination offset in runtime polygon buffer
;   word_2D8AA   = destination segment (same buffer for entire scene)
;   bp           = base vertex index in the shared transformed-vertex tables
;                  (word_2844A; added to each vertex-index field of the poly)
;   word_2D8A0   = address of the 32-byte primary color LUT (in dseg)
;   cx           = number of polygons remaining for this object
;
; Per-polygon flow (8 bytes in → 10 bytes out):
;
;   word 0 (idx1: type/mode bits 15:13, vertex1 bits 10:0):
;     loaded, vertex offset bp added, written verbatim to runtime [di+0].
;
;   word 1 (on disk: color5 bits 15:11, vertex2 bits 10:0):
;     bp added. `cl = ah & 0xF8` captures the 5-bit material color
;     (×8, in bits 7:3). `ah ^= cl` clears those bits, leaving only the
;     vertex2 index. Written to runtime [di+2].
;
;   word 2 (on disk: color5 bits 15:11, vertex3 bits 10:0):
;     Same treatment. `ch = top-5 color ×8`. Color bits cleared.
;     (Empirically, on-disk word 2's color equals word 1's color in 224/224
;      polys inspected across CCERV.POB and CDIAB.POB.)
;     Written to runtime [di+4].
;
;   word 3 (on disk: render-mode5 bits 15:11, vertex4 bits 10:0):
;     Unchanged — `mov [di+6], ax`. The render-mode bits stay in place; they
;     are consumed at draw time by sub_12A82.
;
; Pixel-pair resolution (writes runtime [di+8]):
;
;   ; cx = (ch=w2_color×8, cl=w1_color×8); shift to (w2_color, w1_color):
;   shr cx, 1; shr cx, 1; shr cx, 1
;
;   ; Primary LUT lookup, applied independently to each byte:
;   mov bl, cl; sub bh, bh; add bx, dx; mov cl, [bx]   ; cl = LUT[w1_color]
;   mov bl, ch; sub bh, bh; add bx, dx; mov ch, [bx]   ; ch = LUT[w2_color]
;
;   cmp word_2D628, 13h        ; word_2D628 = current BIOS video mode
;   jz vga_mode_13h            ; ...take the VGA path if mode 13h is active
;
;   ; EGA FALLBACK (word_2D628 != 0x13 — EGA modes 0Dh..12h):
;   ; nibble-mask each byte and emit a solid 16-color pair.
;   and cx, 0F0Fh
;   mov [di+8], cx                                     ; runtime word 4
;
;   ; VGA MODE 13h (word_2D628 == 0x13 — 320x200x256, production path):
;   ;  - If both LUT outputs are < 0x10:
;   ;      bl = (ch<<4) | cl ; bx <<= 1 ; cx = [bx-0x4C3B] = map[0x1F27][bl]
;   ;  - If one or both have bit 4 set (LUT result >= 0x10):
;   ;      lookup via map[0x2127] (16-byte trailer table)
;
; Outputs:
;
;   [di+0] = on-disk word 0 (type/mode + vertex1)
;   [di+2] = vertex2 only (color bits cleared)
;   [di+4] = vertex3 only (color bits cleared)
;   [di+6] = on-disk word 3 unchanged (render mode + vertex4)
;   [di+8] = 16-bit pixel pair (the actual fill color used by rasterizers)
;
; After all polys: word_2AAC2 holds the new polygon count.
;
; Notes
;
;   - This is the canonical "polygon color decoder" we were looking for. It
;     consults THREE tables:
;       * primary 32-byte LUT at word_2D8A0 (rebuilt dynamically by
;         sub_102E8 from day/night/rain flags — essentially identity with
;         per-state overrides for indexes 2, 3, 7, 8, 10)
;       * map[0x1F27] 256-entry pixel-pair table (per-scene)
;       * map[0x2127] 16-byte trailer table (per-scene, secondary)
;
;   - Because on-disk word 1's color == word 2's color empirically, the bx
;     index in VGA mode 13h lands on the *diagonal* of the map[0x1F27] table:
;     0x00, 0x11, 0x22, ..., 0xFF — exactly the entries that hold a "base
;     material color" for the scene rather than a dither pair.
;
;   - The render-mode field in word 3 is independent of the color path; it
;     drives per-frame overrides via sub_12A82 (brake lights, NPC paint,
;     headlights, distance fog) but does not affect the base color resolved
;     here.
